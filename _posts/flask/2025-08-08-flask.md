- GET 请求：从服务器获取数据（通过 URL 查询参数传递）。
- POST 请求：将数据发送到服务器进行处理（如表单提交）。

```py
from flask import Flask, request, render_template

app = Flask(__name__)

@app.route("/", methods=["GET", "POST"])
def form_example():
    if request.method == "POST":
        name = request.form.get("name")  # 获取表单输入
        return f"Hello, {name}!"
    return render_template("form.html")  # 显示 HTML 表单

if __name__ == "__main__":
    app.run(debug=True)
```

### 创建一个表单

创建一个表单，输入一个数字 number，并在表单提交后返回该数字的立方值。

```py
from flask import Flask, request, render_template

app = Flask(__name__)

@app.route("/", methods=["GET", "POST"])
def form_example():
    if request.method == "POST":
        name = request.form.get("name")  # 获取表单输入
        number = request.form.get('number')
        return f"Hello, {name}! Your {number} is {int(number)**3}" # 可以撰写这个输出的html
    return render_template("form.html")  # 显示 HTML 表单

if __name__ == "__main__":
    app.run(debug=True)
```

通过`flask-wtf`

```py
from flask import Flask, request, render_template, redirect, url_for
from flask_wtf import FlaskForm
from wtforms import StringField, SubmitField
from wtforms.validators import DataRequired

app = Flask(__name__)
app.config['SECRET_KEY'] = 'a_very_secret_key_that_should_be_changed'

class NameForm(FlaskForm):
    name = StringField('Name:', validators=[DataRequired()])
    submit = SubmitField('Submit')

@app.route("/", methods=["GET", "POST"])
def form_example():
    form = NameForm()
    if form.validate_on_submit():
        name = form.name.data
        return f"Hello, {name}!"

    return render_template("form_wtf.html", form=form)

if __name__ == "__main__":
    app.run(debug=True)
```

| `form.name.label.text` | 标签的纯文字（不含 HTML）      |
| `form.name()`          | 生成 HTML `<input>` 标签 |
