def auto_agent(self,env):
        """简单贪心策略：蛇头朝着食物走"""
        head_x, head_y = env.snake_x, env.snake_y
        food_x, food_y = env.food_x, env.food_y

        # 优先在 x 方向移动
        if food_x < head_x:
            return 2  # left
        elif food_x > head_x:
            return 3  # right
        elif food_y < head_y:
            return 0  # up
        elif food_y > head_y:
            return 1  # down
        else:
            return random.choice([0, 1, 2, 3])  # 如果重合就随机
